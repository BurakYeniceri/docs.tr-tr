## <a name="example"></a><span data-ttu-id="f5925-111">Örnek</span><span class="sxs-lookup"><span data-stu-id="f5925-111">Example</span></span>

```csharp  
namespace CS1060  
{      
    public class U  
    {  
        public int i;  
    }  
  
    public struct S  
    {  
        public U u;  
        // Add constructor to correct the error.  
        //public S(int val)  
        //{  
        //    u = new U() { i = val };  
        //}  
    }

    {  
        static void Main()  
        {  
            S s;  
            s.u.i = 5;  // CS1060  
  
            //Try these lines instead, and uncomment the constructor in S  
            // S s = new S(0);  
            // s.u.i = 5;  
  
        }  
    }    
}  
```
