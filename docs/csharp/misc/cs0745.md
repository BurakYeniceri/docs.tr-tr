 <span data-ttu-id="c8f2e-109">Aşağıdaki kod CS0745 oluşturur:</span><span class="sxs-lookup"><span data-stu-id="c8f2e-109">The following code generates CS0745:</span></span>

using System.Linq;  
  
public class C  
{  
    public static int Main()
