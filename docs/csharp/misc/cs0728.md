## <a name="example"></a><span data-ttu-id="23dca-110">Örnek</span><span class="sxs-lookup"><span data-stu-id="23dca-110">Example</span></span>

```csharp 
public class ValidBase : IDisposable  
{  
    public void Dispose() {  }  
}  
  
public class Logger  
{  
    public static void dummy()  
    {  
        ValidBase vb = null;  
        using (vb)   
        {  
            vb = null;  // CS0728  
        }  
        vb = null;  
    }  
    public static void Main() { }  
}  
```
