## <a name="example"></a><span data-ttu-id="309fc-108">Örnek</span><span class="sxs-lookup"><span data-stu-id="309fc-108">Example</span></span>

    <Extension()> _  
    Module StringExtensions  
        <Extension()> _

        End Sub  
    End Module
