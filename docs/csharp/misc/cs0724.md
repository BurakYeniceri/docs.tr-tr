```  
// CS0724.cs

{  
    static void Test()  
    {  
        try  
        {  
            throw new Exception();  
        }  
        catch  
        {  
            try  
            {  
            }  
            finally  
            {  
                throw; // CS0724  
            }  
        }  
    }

    {  
    }  
}  
```
