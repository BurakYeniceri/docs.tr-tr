## <a name="example"></a><span data-ttu-id="9cc13-107">Örnek</span><span class="sxs-lookup"><span data-stu-id="9cc13-107">Example</span></span>

public interface I  
{  
    [CLSCompliant(false)]  
    int M();   // CS3010

    }  
}
