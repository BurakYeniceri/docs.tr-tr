## <a name="example"></a><span data-ttu-id="8a941-107">Örnek</span><span class="sxs-lookup"><span data-stu-id="8a941-107">Example</span></span>  
 <span data-ttu-id="8a941-108">Aşağıdaki örnek CS0838 oluşturur:</span><span class="sxs-lookup"><span data-stu-id="8a941-108">The following example generates CS0838:</span></span>

using System.Linq.Expressions;

{  
    class Test  
    {  
        static int Main()  
        {

            Expression<Func<int[,]>> expr2 = () => nums;   
  
            return 1;  
        }  
    }
