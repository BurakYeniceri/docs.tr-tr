## <a name="example"></a><span data-ttu-id="1ab8c-109">Örnek</span><span class="sxs-lookup"><span data-stu-id="1ab8c-109">Example</span></span>

    public partial class C  
    {  
        // Defining declaration.  
        partial void Part();  
  
        // Implementing declaration.

            //...Do something.  
        }

        partial void Part() // CS0757  
        {  
            //...Do something.

        {

        }  
    }
