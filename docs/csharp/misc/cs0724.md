```csharp  
// CS0724.cs  
using System;

{  
    static void Test()  
    {  
        try  
        {  
            throw new Exception();  
        }  
        catch  
        {  
            try  
            {  
            }  
            finally  
            {  
                throw; // CS0724  
            }  
        }  
    }

    {  
    }  
}  
```
